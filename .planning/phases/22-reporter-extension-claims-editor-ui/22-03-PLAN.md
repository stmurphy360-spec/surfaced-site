    ```python
    def compute_sentiment_distribution(
        sentiments: list[dict],
        responses: list[dict],
    ) -> pd.DataFrame:
        """Compute sentiment distribution % by product_line x intent x sentiment.

        Returns DataFrame with columns [product_line, intent, sentiment, pct].
        pct = (count of that sentiment / total responses for that segment) * 100,
        rounded to 1 decimal. All three sentiment values appear for each segment
        even when count is 0 — ensures consistent table shape in render_html().

        SentimentRecord does not carry product_line/intent directly.
        They are joined from responses via prompt_id + model_id.
        Responses are deduplicated on [prompt_id, model_id] before the join
        to avoid inflating counts (responses has 3 rows per prompt for run_number 1/2/3).

        Returns empty DataFrame with correct columns if sentiments is empty.
        """
        COLS = ["product_line", "intent", "sentiment", "pct"]
        if not sentiments:
            return pd.DataFrame(columns=COLS)

        responses_df = pd.DataFrame(responses)
        if responses_df.empty:
            return pd.DataFrame(columns=COLS)

        responses_labels = (
            responses_df[["prompt_id", "model_id", "product_line", "intent"]]
            .drop_duplicates(subset=["prompt_id", "model_id"])
        )

        sentiments_df = pd.DataFrame(sentiments)
        merged = sentiments_df.merge(responses_labels, on=["prompt_id", "model_id"], how="left")
        merged = merged.dropna(subset=["product_line", "intent"])

        all_sentiments = ["positive", "neutral", "negative"]
        records = []
        for (pl, intent), group in merged.groupby(["product_line", "intent"]):
            total = len(group)
            counts = group["sentiment"].value_counts()
            for sent in all_sentiments:
                pct = round((counts.get(sent, 0) / total) * 100, 1) if total > 0 else 0.0
                records.append({
                    "product_line": pl,
                    "intent": intent,
                    "sentiment": sent,
                    "pct": pct,
                })
        return pd.DataFrame(records, columns=COLS)
    ```
