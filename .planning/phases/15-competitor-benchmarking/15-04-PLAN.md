```python
def execute_report(run_id: str) -> None:
    responses = storage.read_responses(run_id)
    extractions = storage.read_extractions(run_id)
    scores = storage.read_scores(run_id)

    try:
        run_meta = storage.read_run_meta(run_id)
    except FileNotFoundError:
        run_meta = {"run_id": run_id, "run_date": "unknown", "models": {}, "total_prompts": "?"}

    visibility_df = compute_visibility(extractions, responses)
    rank_df = compute_rank_distribution(extractions, responses)

    if extractions:
        ideal_df = compute_ideal_content(extractions, scores, responses)
    else:
        ideal_df = pd.DataFrame(columns=["product_line", "model_id", "ideal_pct"])

    product_line_labels = _load_product_line_labels()
    visibility_path, ideal_path = write_csvs(run_id, visibility_df, ideal_df, product_line_labels)

    html = render_html(run_meta, visibility_df, rank_df, ideal_df, product_line_labels)
    report_path = storage.run_dir(run_id) / "report.html"
    report_path.write_text(html, encoding="utf-8")

    print(f"Report: {report_path}")
    print(f"Visibility CSV: {visibility_path}")
    print(f"Ideal Content CSV: {ideal_path}")
```
